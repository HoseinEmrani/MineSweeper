1 /** create new game
  2  * @param {jQueryObject} map css selector to build minesweepr map in
  3  * @param {int} x number of columns in the map
  4  * @param {int} y number of rows in the map
  5  * @param {int} mineNumber number of mines that tile contains
  6  * @return {mapData} return data of map created
  7  * @namespace make minesweeper map and set random mines
  8  */
  9 var mineMap = function( map , x , y , mineNumber ) {
 10 	 /**build html code of 'ul' childs with the given data
 11 	 * @function
 12 	 * @param {int} i data-row value of li tag
 13 	 * @return {String} a 'li' html code
 14 	 */
 15 	var buildRows =function(i) { return "<li data-row=\"" + i + "\"></li>"; };
 16 	
 17 	/**build html code of 'li' childs with the given data
 18 	 * @function
 19 	 * @param {int} i data-cell value of span tag
 20 	 * @return {String} a 'span' html code
 21 	 */
 22         var buildCell =function(i) { return "<span data-cell=\"" + i + "\"></span>";};
 23 	
 24 	/**select a cell by it's html data value
 25 	 * @function
 26 	 * @param {int} i data-cell value of span tag
 27 	 * @param {int} j data-row value of li tag
 28 	 * @return {String} a 'span' html code
 29 	 */
 30         var mineCell = function(i,j) {return $("[data-row=" + j + "]").children("[data-cell=" + i + "]")};
 31 	
 32 	/**if true,clicks on the map do nothig
 33 	 * @type boolean
 34 	 */
 35         var IS_FINISHED = false;
 36 
 37 	/**number of cells in the map
 38 	 * @type int
 39 	 */
 40         var notShownNumber = x*y;
 41 
 42 	/**number of flags player used
 43 	 * @type int
 44 	 */
 45         var flagCounter = 0;
 46 
 47 	/**number of flags in currect position
 48 	 * @type int
 49 	 */
 50         var correctFlag = 0;
 51 	/** @class saving map data here
 52 	 *
 53 	 * 
 54 	 * 
 55 	 */
 56 	var mapData=function(x,y,mineNumber){
 57 		this.X = x;
 58 		this.Y = y;
 59 		this.mineNumber=mineNumber;
 60 	};
 61 
 62 	// Define Win Function 
 63 	/**make changes when player wins
 64 	 * @function win
 65 	 * @return nothing
 66 	 */
 67 	var win = function() {
 68 	    // Set gmae state tu finished.
 69 	    IS_FINISHED = true;
 70 	    GAME_RUNNING = false;
 71 	    timerPause();
 72 	    alert("Yesssss! You Win!!!!!");
 73 	}
 74 
 75 	// Define Lose Function
 76 	/**make changes when player loses
 77 	 * @function lose
 78 	 * @return nothing
 79 	 */
 80 	var lose = function() {
 81 	    // Set gmae state tu finished.
 82 	    IS_FINISHED = true;
 83 	    GAME_RUNNING = false;
 84 	    timerPause();
 85 	    // Check for correct or wrong flags
 86 	    checkFlag();	    
 87 	    alert("NAN NA NANA NA! You Lose :D !!!!!");
 88 	}
 89 	
 90 	// Reset The Timer
 91 	GAME_RUNNING = false;
 92 	timerReset();	    
 93 	
 94 	/**number of mines around to a cell or -1 if it's a mine.<br>
 95 	 * This array filled with random mines and then we count number of mines around the cells not contain mine
 96 	 * @type {array[][]}
 97 	 * @field
 98 	 */
 99 	var mineArea = new Array(x);
100 	for ( var i =0; i< x; i++) {
101 	    mineArea[i] = new Array(y);
102 	}
103 	
104 	// Building The Map
105 
106 	$(map).html("");
107     
108 	for(var i = 0; i < y ; i++ ){
109 		$(map).append(buildRows(i));
110 	}
111     
112     for(var j =0; j < x; j++) {
113         $(map).children('li').append(buildCell(j));
114     }
115     if(x > y){
116         $('span').css({'width':500/x + 'px','line-height':500/x +'px','height':500/x + 'px'});
117     }else{
118         $('span').css({'width':400/y + 'px','line-height':400/y +'px','height':400/y + 'px'});
119     }
120     
121     
122     var count = 0;
123     while( count < mineNumber ) {
124         
125         var randX = Math.floor( Math.random() * x ),
126             randY = Math.floor( Math.random() * y );
127             if ( mineArea[randX][randY] != -1 ) {
128                 mineArea[randX][randY] = -1;
129                 count++;
130             }
131         
132     }
133     
134     // counting the numbers of mine beside a cell.
135     for ( var i =0 ; i < x; i++)  {
136         for(var j=0; j < y; j++) {
137             
138             if(mineArea[i][j] == -1) continue;
139             mineArea[i][j] = 0;
140             var sum =0;
141             
142             for(var ii = i-1; ii <= i+1 ; ii++) {
143                 for (var jj = j-1; jj <= j + 1; jj++) {
144                     if(ii>=0 && jj>=0 && ii<x && jj<y){
145                         if(mineArea[ii][jj]==-1){
146                             sum++;
147                         }
148                     }
149                 }            
150             }
151             
152             mineArea[i][j] = sum;
153         }
154     }
155 
156     // Define Click Function
157     /**any changes should be take place when player click on a cell
158      * @event
159      * @param {cssSelector} clicked span that player clicked on
160      * @return nothing
161      */
162     var spanClick = function(clicked) {
163         if( IS_FINISHED ) return;
164         if ( $(clicked).hasClass('flag') ) return;
165         if ( !GAME_RUNNING ) GAME_RUNNING = true;
166         var spanRow = $(clicked).parent("li").data("row"),
167             spanCell = $(clicked).data("cell");
168         
169             if( mineArea[spanCell][spanRow] == -1 ) {
170                 $(clicked).addClass("mineFired");
171                 for ( var i =0 ; i < x; i++)  {
172                     for(var j=0; j < y; j++) {
173                         if( mineArea[i][j] == -1 ) {
174                             $(mineCell(i,j)).addClass("mineShow");
175                         }
176                     }
177                 }
178                 lose();
179             }
180             
181             else if( !( $(clicked).hasClass("Show")) ) {
182                 $(clicked).addClass('Show');
183                 notShownNumber--;
184                 
185                 $(clicked).text( mineArea[spanCell][spanRow]==0 ? " " : mineArea[spanCell][spanRow]);
186                 
187                 if(mineArea[spanCell][spanRow]==0 ){
188                     for(var i = spanCell-1; i <= spanCell+1 ; i++) {
189                         for (var j = spanRow -1; j <= spanRow + 1; j++) {
190                             if(i>=0 && j>=0 && i<x && j<y){
191                                 spanClick($(mineCell(i,j)));
192                             }
193                         }            
194                     }
195                 }
196             }
197             if ( $('.Show').length + mineNumber == x*y || notShownNumber==mineNumber) {
198 				if( IS_FINISHED ) return;
199 				win();
200 				return; 
201 			}
202     }
203     
204     // Define Flag Function
205     /**Flages a span if doesn't have or remove it if have!
206      * @event
207      * @param {cssSelector} clicked span that player right clicked on
208      * @return nothing
209      */
210     var spanFlag = function(clicked) {
211         if( IS_FINISHED ) return;
212         var spanRow = $(clicked).parent("li").data("row"),
213             spanCell = $(clicked).data("cell");
214         
215         if ($(clicked).hasClass("Show") ) return;
216         
217         if( $(clicked).hasClass("flag")) {
218            $(clicked).removeClass('flag');
219             flagCounter --;
220             if( mineArea[spanCell][spanRow] == -1 ) {
221                 correctFlag--;
222             }
223         
224             if ( correctFlag == flagCounter && flagCounter == mineNumber ) {
225                 win();
226             }
227             
228             return;
229         }
230            
231        $(clicked).addClass('flag');
232 
233         flagCounter ++;
234         
235         if( mineArea[spanCell][spanRow] == -1 ) {
236             correctFlag++;
237         }
238 
239         if ( correctFlag == flagCounter && flagCounter == mineNumber ) {
240             win();
241         }
242     
243     }
244     
245     /**When player loses,this function check wich cell that player marks as flaged,are currect and wich not
246      * @function
247      * @return nothing
248      */
249     var checkFlag = function() {
250         $('.flag').each(function() {
251             var spanRow = $(this).parent("li").data("row"),
252             spanCell = $(this).data("cell");
253             
254             if( mineArea[spanCell][spanRow] == -1 ) {
255                 $(this).addClass('correct-flag');
256             }
257             
258             else {
259                 $(this).addClass('wrong-flag');
260             }
261             
262         });
263     }
264     
265     // Fire Click Functions
266     $(map).children("li").children("span").mousedown(function(e){
267         if( e.button == 2 || e.button == 3 ) {
268             spanFlag($(this));
269         }
270         else {
271             spanClick($(this));
272         }
273     });
274     
275       
276     return new mapData(x,y,mineNumber);
277     
278 }
